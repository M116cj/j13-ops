def _query_view():
    """Returns (deployable_count:int, age_h:float|None) or raises."""
    r = subprocess.run(
        [
            "docker", "exec", "deploy-postgres-1",
            "psql", "-U", "zangetsu", "-d", "zangetsu",
            "-t", "-A", "-F", "|",
            "-c", "SELECT deployable_count, last_live_at_age_h FROM zangetsu_status",
        ],
        capture_output=True, text=True, timeout=PSQL_TIMEOUT_S,
    )
    if r.returncode != 0:
        raise RuntimeError(f"psql exit {r.returncode}: {r.stderr.strip()[:200]}")
    line = r.stdout.strip().split("\n")[0]
    parts = line.split("|")
    if len(parts) < 2:
        raise ValueError(f"malformed VIEW row: {line!r}")
    deployable = int(parts[0])
    age_str = parts[1].strip()
    age_h = float(age_str) if age_str else None
    return deployable, age_h
